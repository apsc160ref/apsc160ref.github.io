Question 1

Write a function int stringLength(char str[]); that computes and returns the length of the supplied string. The equivalent function in <string.h> is strlen(...).

int stringLength(char str[]){
    int stringLength = 0;
    for( int character = 0; str[character] != '\0' ; character++ ){
        stringLength++;
    }
    return stringLength;
}

Question 2

Write a function void stringCopy(char dest[], char src[]); that computes and returns the length of the supplied string. The equivalent function in <string.h> is strcpy(...).

void stringCopy(char dest[], char src[]){
    int characterIndex = 0;
    for ( characterIndex = 0 ; src[characterIndex] != '\0' ; characterIndex++ ) {
        dest[characterIndex] = src[characterIndex];
    }
    dest[characterIndex++] = '\0';
}

Question 3

Write a function void stringAppend(char dest[], char src[]); that computes and returns the length of the supplied string. The equivalent function in <string.h> is stringcat(...).

void stringAppend(char dest[], char src[]){
    int concatenateStart = stringLength(dest);
    int i = 0;
    for ( i = 0 ; src[i] != '\0' ; i++ ){
        dest[concatenateStart + i] = src[i];
    }
    dest[concatenateStart + (i++)] = '\0'; // terminating null character
}

Example 1: Assigning / Extracting a String and a Number from a formatted string

char str[10] = "";
int num = 0;
char input[] = "testString 22";
sscanf(input, "%s %d", str, &num);

Value of str: "testString", Value of num: 22

Example 2: Assigning / Extracting two strings from a formatted string

char str1[6] = "";
char str2[6] = "";
char input2[] = "catch 22";
sscanf(input2, "%s %s", str1, str2);

Value of str1: "catch", Value of str2: "22" ← THIS IS NOT AN INT

char str3[15] = "";
