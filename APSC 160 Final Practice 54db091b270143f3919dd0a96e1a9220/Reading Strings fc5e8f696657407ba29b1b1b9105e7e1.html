Reading Strings

Goal: Read strings from different sources, and assign them to character arrays. Strings will have spaces like normal sentences.

Our three tasks for each section are as follows:

  • Read string with spaces from scanf
  • Read string with spaces from FILE*

Section 1 ⇒ scanline modifications

/*
SUPER IMPORTANT SCANLINE FUNCTIONS
*/

/* scans a line from standard input */
int scanline(char line[])
{
    int len = 0; /* length of line */
    char c = '\0'; /* next character to read */
    int success = 0; /* success of scanf */
    success = scanf("%c", &c);
    while (success == 1 && c != '\n') {
        line[len++] = c; /* copy character */
        success = scanf("%c", &c); /* next character */
    }
    line[len] = '\0'; /* terminating NUL */
    return len;
}

/* scans a line from a file */
int fscanline(FILE* file, char line[])
{
    int len = 0; /* length of line */
    char c = '\0'; /* next character to read */
    int success = 0; /* success of fscanf */
    
    /* loop until end of line or file */
    success = fscanf(file, "%c", &c);
    while (success == 1 && c != '\n') {
        line[len++] = c; /* copy character */
        success = fscanf(file, "%c", &c); /* next character */
    }
    line[len] = '\0'; /* terminating NUL */
    return len;
}

SCANLINE → SCANF TO STRING (SPACES WORK)

Reading String(s) from FILE* using fgets 

int main ( void ){
    
    char string[250]; // char string on each line
    int lineCount = 0;
    
    FILE* inFile = fopen(DATA_FILE, "r");
       
    if (inFile == NULL)
        printf("Error opening input file - program terminating...\n");

        while ( fgets(string, 250, inFile) ){ // returns true if scan is successful

Getting digits from string:

#include <stdio.h>
#include <stdlib.h>
#include <strings.h>
#include <math.h>
#define DATA_FILE "numberString.txt"

int main ( void ){
    
    char stringNumber[250]; // char string on each line
    int currentNumber = 0;
    int numbers[10];
    
    FILE* inFile = fopen(DATA_FILE, "r");
       
    if (inFile == NULL)
        printf("Error opening input file - program terminating...\n");
    else {
        while ( fgets(stringNumber, 250, inFile) ){
            //before we turn the string into a number, first we get the length
            int length = strlen(stringNumber) - 1;
            
            // now we convert the string into a number using sscanf
            sscanf( stringNumber, "%d", &currentNumber );
            
            int numbersIndex = 0;
            for ( int power = length - 1 ; power >= 0 ; power-- ){
                int frac = (int) currentNumber / pow(10, power);
                numbers[numbersIndex] = frac;
                currentNumber -= frac * pow(10, power);
                numbersIndex++;
            }
            for (int i = 0; i < 4; i++ ){
                printf("%d ", numbers[i]);
            }
        }
        fclose(inFile); // close the file
        printf("\n");
    }
    return 0;

}
